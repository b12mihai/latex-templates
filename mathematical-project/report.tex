\begin{equation}
\label{eq:SDN}
\left\{ \begin{aligned}
        x'(t) = & f(t, x(t), u(t)) \ \\
        y(t) = & g(t, x(t)) \ 
       \end{aligned}
 \right.
 \qquad 
\end{equation}
